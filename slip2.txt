const http = require("http");
const fs = require("fs");
const url = require("url");

http.createServer((req, res) => {
    let q = url.parse(req.url, true).query;

    if (q.f1 && q.f2) {
        fs.appendFileSync(q.f2, fs.readFileSync(q.f1));
        res.end("Appended Successfully");
    } else {
        res.end(`<form>
            F1:<input name="f1">
            F2:<input name="f2">
            <input type="submit">
        </form>`);
    }
}).listen(3000);s
