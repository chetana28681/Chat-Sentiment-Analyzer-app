const http = require("http");
const fs = require("fs");
const url = require("url");

http.createServer((req, res) => {

    // Get filename from URL
    let q = url.parse(req.url, true);
    let filename = "." + q.pathname;

    // Read file
    fs.readFile(filename, (err, data) => {
        if (err) {
            // File not found → 404 error
            res.writeHead(404, { "Content-Type": "text/html" });
            return res.end("404 File Not Found");
        }

        // Success → send file content
        res.writeHead(200, { "Content-Type": "text/html" });
        res.write(data);
        return res.end();
    });

}).listen(3000);

console.log("Server running at http://localhost:3000");

Create a node js file named main.js for event-driven application. There should be a main loop
that listens for events, and then triggers a callback function when one of those events is
detected.

const events = require('events');

// create event object
const eventEmitter = new events.EventEmitter();

// callback function
const myEventHandler = () => {
    console.log("Event detected! Callback function executed.");
};

// register event
eventEmitter.on('start', myEventHandler);

// main loop (simulated event trigger)
console.log("Waiting for events...");

setTimeout(() => {
    eventEmitter.emit('start');   // trigger event
}, 3000);

slip 9 a
Create a Node.js file that writes an HTML form, with a concatenate two string

const http = require("http");
const url = require("url");

http.createServer((req, res) => {

    let q = url.parse(req.url, true).query;
    let r = q.s1 ? (q.s1 + q.s2) : "";

    res.writeHead(200, { "Content-Type": "text/html" });

    res.end(`
        <form>
            S1:<input name="s1">
            S2:<input name="s2">
            <input type="submit">
        </form>
        <h3>${r}</h3>
    `);

}).listen(3000);

console.log("Server running on 3000");



slip 9 b

Q.2) Create a Node.js file that opens the requested file and returns the content to the client
 If anything goes wrong, throw a 404 error
 

const http = require("http");
const fs = require("fs");

http.createServer((req, res) => {

    let file = req.url.slice(1);

    fs.readFile(file, (err, data) => {
        if (err) {
            res.writeHead(404, { "Content-Type": "text/html" });
            res.end("Error 404: File Not Found");
        } else {
            res.writeHead(200, { "Content-Type": "text/html" });
            res.end(data);
        }
    });

}).listen(3000);

console.log("Server running on 3000");

Create a Node.js file that will convert the output "HELLO WORLD!" into lower-case letters.
[10 Marks]


const http = require("http");

http.createServer((req, res) => {

    let text = "HELLO WORLD!";
    let result = text.toLowerCase();

    res.writeHead(200, { "Content-Type": "text/html" });
    res.end("Lowercase Output: " + result);

}).listen(3000);

(use ng-view)



<!DOCTYPE html>
<html ng-app="app">
<head>
<script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.2/angular.min.js"></script>
<script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.2/angular-route.js"></script>

<a href="#!ds">DS</a> |
<a href="#!dbms">DBMS</a> |

app.config(function($routeProvider){

    .when("/dbms",{template:"<h3>DBMS</h3>"})
    .when("/java",{template:"<h3>Java</h3>"})

});
</script>
